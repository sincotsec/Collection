\section{$\omega$-функции}

\begin{equation*}\begin{aligned}
\omega_{0}(x) &= \frac{x^0}{0!} \\
\omega_{1}(x) &= \frac{x^1}{1!} \\
\omega_{2}(x) &= \frac{x^2}{2!} \\
\omega_{3}(x) &= \frac{x^3}{3!} \\
\omega_{4}(x) &= \frac{x^4}{4!} \\
&\ldots \\
\end{aligned}\end{equation*}

\begin{equation*}\begin{aligned}
\omega_{-1}(x) &= \frac{x^{-1}}{(-1)!} \\
\omega_{-2}(x) &= \frac{x^{-2}}{(-2)!} \\  
\omega_{-3}(x) &= \frac{x^{-3}}{(-3)!} \\
\omega_{-4}(x) &= \frac{x^{-4}}{(-4)!} \\
&\ldots \\
\end{aligned}\end{equation*}
