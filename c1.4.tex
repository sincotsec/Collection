\section{Фундаментальные функции от $ 1 + x $}

\subsection{Нулевой порядок}

\begin{equation*}
\begin{aligned}
\frac{(1 + x)^0}{0!} &=
  \frac{1}{0!0!} 
\\
\frac{(1 + x)^1}{1!} &=
  \frac{1}{1!0!} 
+ \frac{1}{0!1!} x
\\
\frac{(1 + x)^2}{2!} &= 
  \frac{1}{2!0!} 
+ \frac{1}{1!1!} x 
+ \frac{1}{0!2!} x^2
\\
\frac{(1 + x)^3}{3!} &=
  \frac{1}{3!0!} 
+ \frac{1}{2!1!} x 
+ \frac{1}{1!2!} x^2 
+ \frac{1}{0!3!} x^3
\\
\frac{(1 + x)^4}{4!} &=
  \frac{1}{4!0!} 
+ \frac{1}{3!1!} x 
+ \frac{1}{2!2!} x^2 
+ \frac{1}{1!3!} x^3 
+ \frac{1}{0!4!} x^4
\\
\frac{(1 + x)^5}{5!} &=
  \frac{1}{5!0!} 
+ \frac{1}{4!1!} x 
+ \frac{1}{3!2!} x^2 
+ \frac{1}{2!3!} x^3 
+ \frac{1}{1!4!} x^4
+ \frac{1}{0!5!} x^5
\\
&\ldots
\\
\end{aligned}
\end{equation*}
